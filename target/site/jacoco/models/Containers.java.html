import javax.persistence.*;
//CHECKSTYLE:OFF

@Data
@NoArgsConstructor
@AllArgsConstructor
@Builder
@Getter

@Entity

public class Containers {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Integer Id;

    private Integer Container;

    private String Type;

    private Integer Amount;

    private String WineId;

}
